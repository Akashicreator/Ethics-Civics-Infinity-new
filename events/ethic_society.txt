namespace=ethic_society

country_event = { # 社会状态更新（国家月检）
    id = ethic_society.0
    is_triggered_only = yes
    hide_window = yes
    immediate = {
		every_owned_planet = {
			if = {
				limit = {
					AND = {
						is_colony = yes
						is_under_colonization = no
					}
				}
				planet_event = { id = ethic_society.1 }
			}
		}
    }
}

planet_event = { # 社会状态更新（星球月检与触发）
	id = ethic_society.1
    is_triggered_only = yes
    hide_window = yes
    immediate = {
		set_variable = { which = crime_rate_sub value = value:crime_rate_sub }
		
		if = {
			limit = { NOT = { has_deposit = d_ethic_society } }
			add_deposit = d_ethic_society
		}
    }
}
