add_vanguard_pop = {
	random_list = {
		100 = {}
		$POSSIBILITY$ = {
			random_owned_pop = {
				limit = { NOT = { has_trait = trait_hive_mind } }
				kill_pop = yes
			}
			every_owned_species = { 
				limit = { has_trait = trait_unnatural } 
				save_event_target_as = forever_species 
			}
			create_pop = {
				species = event_target:forever_species
			}
			change_variable = { which = vanguard_num value = 0.33 }
			if = {
				limit = { NOT = { has_country_flag = vanguard_no_window } }
				owner = { country_event = { id = ethic_civic_yearly.4 } }
			}
		}
	}
}

graduated_from_university = {
	random_list = {
		100 = {}
		$POSSIBILITY$ = {
			random_owned_pop = {
				limit = { has_modifier = pop_uneducated }
				if = {
					limit = { owner = { NOT = { has_policy_flag = private_education_eo } } }
					remove_modifier = "pop_uneducated"
					add_modifier = {
						modifier = "pop_highly_educated"
						days = -1
					}
					planet = { change_variable = { which = secondary_education_rate value = 0.10 } }
				}
				if = {
					limit = {
						owner = { has_policy_flag = private_education_eo }
						OR = {
							has_living_standard = { type = living_standard_stratified }
							has_living_standard = { type = living_standard_subsistence }
							has_living_standard = { type = living_standard_chemical_bliss }
						}
					}
					remove_modifier = "pop_uneducated"
					add_modifier = {
						modifier = "pop_highly_educated"
						days = -1
					}
					planet = { change_variable = { which = secondary_education_rate value = 0.33 } }
				}
				if = {
					limit = {
						owner = { has_policy_flag = private_education_eo }
						OR = {
							has_living_standard = { type = living_standard_normal }
							has_living_standard = { type = living_standard_good }
							has_living_standard = { type = living_standard_shared_burden }
							has_living_standard = { type = living_standard_utopian }
							has_living_standard = { type = living_standard_decadent }
							has_living_standard = { type = living_standard_academic_privilege }
						}
					}
					remove_modifier = "pop_uneducated"
					add_modifier = {
						modifier = "pop_highly_educated"
						days = -1
					}
					planet = { change_variable = { which = secondary_education_rate value = 0.10 } }
				}
				if = {
					limit = {
						owner = { has_policy_flag = private_education_eo }
						OR = {
						}
					}
					remove_modifier = "pop_uneducated"
					add_modifier = {
						modifier = "pop_highly_educated"
						days = -1
					}
					planet = { change_variable = { which = secondary_education_rate value = 0.07 } }
				}
				if = {
					limit = {
						owner = { has_policy_flag = private_education_eo }
					}
					remove_modifier = "pop_uneducated"
					add_modifier = {
						modifier = "pop_highly_educated"
						days = -1
					}
					planet = { change_variable = { which = secondary_education_rate value = 0.05 } }
				}

graduated_from_secondary_school = {
	random_list = {
		100 = {}
		$POSSIBILITY$ = {
			random_owned_pop = {
				if = {
					limit = {
						has_modifier = pop_uneducated
						OR = {
							is_enslaved = no
							has_slavery_type = { country = owner type = slavery_indentured }
						}
					}
					remove_modifier = "pop_uneducated"
					add_modifier = {
						modifier = "pop_educated"
						days = -1
					}
				}
				if = {
					limit = {
						has_modifier = pop_uneducated
						is_enslaved = yes
						NOT = { has_slavery_type = { country = owner type = slavery_indentured } }
					}
					remove_modifier = "pop_uneducated"
					add_modifier = {
						modifier = "pop_slave_educated"
						days = -1
					}
				}
			}
		}
	}
}
