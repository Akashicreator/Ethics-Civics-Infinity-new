			add = 100000

			is_primitive = yes
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	format = "{AB{[This.Capital.System.GetName] <soviet_union>}}"
	 
	noun = format.species
	prefix_format = format.coa.1

}
