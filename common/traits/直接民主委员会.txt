    custom_subtitle = councilor_subclass_admiral
    subclass_trait = yes
    councilor_trait = yes
    inline_script = {
        script = trait/icon_subclass
        SUBCLASS = strategist
        COUNCIL = yes
    }
    custom_tooltip_with_modifiers = councilor_subclass_tt
    triggered_modifier = {
        potential = {

    }
    leader_potential_add = {
        NAND = {

        }
    }
    leader_class = { admiral }
    subclass_background_icon = "GFX_leader_subclass_strategist_background"
}
