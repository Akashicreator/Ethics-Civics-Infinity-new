leader_trait_bureaucrat = { # Unifier
	inline_script = {
		script = trait/icon
		CLASS = governor
		ICON = "GFX_leader_trait_bureaucrat"
		RARITY = common
		COUNCIL = no
		TIER = 1
	}
	modifier = {
		planet_bureaucrats_upkeep_mult = -0.10
	}
	on_gained_effect = {
		add_age = value:tenth_of_leader_lifespan
		random_list = {
			1 = { change_background_job = bureaucrat }
			1 = { change_background_job = politician }
			1 = { change_background_job = government_employee }
			1 = { change_background_job = clerk }
		}
	}
	leader_potential_add = {
		is_pool_leader = yes # Starting trait only.
		is_ruler = no
	}
	leader_class = { governor }
	selectable_weight = @class_trait_weight
}

			}
		}
	}
	inline_script = {
		script = trait/icon
		CLASS = leader
		ICON = "GFX_leader_trait_capitalist"
		RARITY = common
		COUNCIL = no
		TIER = 1
	}
	inline_script = {
		script = paragon/leader_resource_produces
		RESOURCE = energy
		AMOUNT = 4
	}
	leader_class = { admiral scientist general governor }
	opposites = {
		leader_trait_lavish_lifestyle
		leader_trait_lavish_lifestyle_2
	}
	selectable_weight = {
		weight = @shared_trait_weight
		inline_script = {
			script = "paragon/ethic_weight_mult"
			IS = materialist
			NOT = spiritualist
		}
		inline_script = {
			script = "paragon/existing_trait_weight_mult"
			TRAIT = leader_trait_capitalist
		}
	}
}

		COUNCIL = triggered
		TIER = none
	}
	triggered_modifier = {
		potential = {
			is_ruler = no

	triggered_modifier = {
		potential = {
			is_ruler = no

	}
	triggered_self_modifier = {
		potential = {
			is_ruler = no
			leader_class = scientist
		}
		assist_research_mult = 0.33
	}
	triggered_self_modifier = {
		potential = {
			is_ruler = no
			leader_class = scientist

		leader_scientists_society_research_produces_add = 2.5
		leader_scientists_engineering_research_produces_add = 2.5
		mult = trigger:has_skill
	}
	triggered_modifier = {
		potential = {

	triggered_modifier = {
