leader_trait_bureaucrat = { # Unifier
	inline_script = {
		script = trait/icon
		CLASS = governor
		ICON = "GFX_leader_trait_bureaucrat"
		RARITY = common
		COUNCIL = no
		TIER = 1
	}
	modifier = {
		planet_bureaucrats_upkeep_mult = -0.10
	}
	on_gained_effect = {
		add_age = value:tenth_of_leader_lifespan
		random_list = {
			1 = { change_background_job = bureaucrat }
			1 = { change_background_job = politician }
			1 = { change_background_job = government_employee }
			1 = { change_background_job = clerk }
		}
	}
	leader_potential_add = {
		is_pool_leader = yes # Starting trait only.
		is_ruler = no
	}
	leader_class = { governor }
	selectable_weight = @class_trait_weight
}
