aicbm_exotic_gases_expenditure_ships = {
	resource = exotic_gases
	type = expenditure
	category = ships

	potential = {
		has_resource = { 
			type = exotic_gases 
			amount > 800
		}
	}

	weight = {
		weight = 1

		# spend more exotic_gases building ships while at war
		modifier = {
			factor = 2
			is_at_war = yes
		}

		# spend less exotic_gases on ships if over capacity
		modifier = {
			factor = 0.5
			used_naval_capacity_percent >= 1.0
		}
	}
}
