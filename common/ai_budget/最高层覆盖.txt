aicbm_unity_expenditure_missing_scientist = {
	resource = unity
	type = expenditure
	category = leaders

	potential = {
		has_resource = { 
			type = unity 
			amount > 8000
		}
		OR = {
			scientist_count < 20

			NOT = {
				research_leader = {
					area = physics
				}
			}
			NOT = {
				research_leader = {
					area = society
				}
			}
			NOT = {
				research_leader = {
					area = engineering
				}
			}
		}
	}

	weight = {
		weight = 1.6
	}

	desired_max = {
		base = 2.5
		mult = value:leader_cost #will cap at 25% more than it costs to buy a leader
	}
}
