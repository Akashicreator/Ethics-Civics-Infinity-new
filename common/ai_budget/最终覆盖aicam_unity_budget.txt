		}

	}

	}

aicbm_unity_expenditure_ships = {
	resource = unity
	type = expenditure
	category = ships

	potential = {
		has_resource = { 
			type = unity 
			amount > 8000
		}
	}

	weight = {
		weight = 1

		# spend more unity building ships while at war
		modifier = {
			factor = 2
			is_at_war = yes
		}

		# spend less unity on ships if over capacity
		modifier = {
			factor = 0.5
			used_naval_capacity_percent >= 1.0
		}
	}
}
